<head>
<title>HTML5</title>
<script src="jquery-latest.min.js"></script>
<script type="text/javascript">
//BEGIN LIBRARY CODE
var x = 150;
var y = 150;
var dx = 2;
var dy = 4;
var ctx;
var paddleh = 10;
var WIDTH = 300;
var HEIGHT= 300;
var paddley = HEIGHT-paddleh;
var paddlew = 75;
var intervalId = 0;
var rightDown = false;
var leftDown = false;
var paddlemovevertical = 'stop';
var paddle_vertical_speed ;
function $(id) {
return document.getElementById(id);
}
function init() {
ctx = $('canvas').getContext('2d');
paddlex = WIDTH / 2;
intervalId = setInterval(draw, 10);
return intervalId;
}
function circle(x,y,r) {
ctx.beginPath();
ctx.arc(x, y, r, 0, Math.PI*2, true);
ctx.closePath();
ctx.fill();
}
function rect(x,y,w,h) {
ctx.beginPath();
ctx.rect(x,y,w,h);
ctx.closePath();
ctx.fill();
}
function clear() {
ctx.clearRect(0, 0, WIDTH, HEIGHT);
}
function onKeyDown(evt) {
if (evt.keyCode == 39)
paddlex += 30;
//rightDown = true;
else if (evt.keyCode == 37)
paddlex -= 30;
//upkey = true; hierdie moet check if on floor.
else if (evt.keyCode == 38 && paddley == HEIGHT-paddleh)
{
paddlemovevertical = 'up';
paddle_vertical_speed = 15;
}
}
window.addEventListener('keydown',onKeyDown,true)
function draw() {
clear();
circle(x, y, 10);
rect(paddlex, paddley, paddlew, paddleh);
//===============================================
//padleMovement
if (paddlemovevertical == 'up')
{
paddle_vertical_speed -= 1;
}
else if (paddlemovevertical == 'down')
{
paddle_vertical_speed += 1;
}
if (paddley <= 100) // remove
{
paddlemovevertical = 'down';
}
else if (paddley > HEIGHT-paddleh)
{
paddley = HEIGHT-paddleh;
paddlemovevertical = 'stop';
}
if (paddlemovevertical == 'up')
{
paddley -= paddle_vertical_speed ;
}
else if (paddlemovevertical == 'down')
{
paddley += paddle_vertical_speed;
}
//===============================================
//if ball hits wall
if (x + dx > WIDTH || x + dx < 0)
{
dx = -dx;
}
//if ball hits roof
if (y + dy < 0)
{dy = -dy;}
//if ball hits paddle
else if (x > paddlex && x < paddlex + paddlew && y > paddley)
{dy = -dy;}
else if (y + dy > HEIGHT) {
{ clearInterval(intervalId); }
}
x += dx;
y += dy;
}
init();
</script>
